<script>
  // URL Kontrolü (Dil Tespiti)
  var path = window.location.pathname;
  var isEn = path.indexOf('/en/') !== -1;
  
  // Daktilo Efekti İçin Mesajlar
  var messages = {
    tr: [
      "> BAĞLANTI KURULUYOR...",
      "> HATA: 404 - Sinyal Kaybedildi.",
      "> Aradığınız sayfa uzay boşluğunda kaybolmuş olabilir.",
      "> Veya taşınmış olabilir.",
      "> ÖNERİ: <a href='/'>Ana Üsse Dön</a>"
    ],
    en: [
      "> ESTABLISHING CONNECTION...",
      "> ERROR: 404 - Signal Lost.",
      "> The page you are looking for might be lost in space.",
      "> Or it might have been moved.",
      "> SUGGESTION: <a href='/en/'>Return to Base</a>"
    ]
  };

  var content = isEn ? messages.en : messages.tr;
  var outputDiv = document.getElementById('output');
  var lineIndex = 0;
  var charIndex = 0;
  var currentHTML = "";

  function typeWriter() {
    if (lineIndex < content.length) {
      var line = content[lineIndex];
      
      // Eğer satır HTML içeriyorsa (Link gibi), direkt bas
      if (line.includes("<")) {
        currentHTML += line + "<br><br>";
        outputDiv.innerHTML = currentHTML;
        lineIndex++;
        setTimeout(typeWriter, 500);
      } 
      // Düz metinse harf harf yaz
      else {
        if (charIndex < line.length) {
          outputDiv.innerHTML = currentHTML + line.substring(0, charIndex + 1);
          charIndex++;
          setTimeout(typeWriter, 30); // Yazma hızı
        } else {
          currentHTML += line + "<br>";
          outputDiv.innerHTML = currentHTML;
          lineIndex++;
          charIndex = 0;
          setTimeout(typeWriter, 300); // Satır arası bekleme
        }
      }
    }
  }

  // Animasyonu Başlat
  setTimeout(typeWriter, 500);
</script>
